SELECT 
    -- Identificação da unidade de saúde
    FAI.CO_DIM_UNIDADE_SAUDE_1,
    DUS.NU_CNES,
    DUS.NO_UNIDADE_SAUDE,
    
    -- Identificação da equipe
    FAI.CO_DIM_EQUIPE_1,
    DE.NU_INE,
    DE.NO_EQUIPE,
    
    -- Identificação do profissional
    FAI.CO_DIM_PROFISSIONAL_1,
    COALESCE(TP.CO_SEQ_PROF, DP.CO_SEQ_DIM_PROFISSIONAL) AS NU_PROFISSIONAL,
    COALESCE(UPPER(TP.NO_PROFISSIONAL_FILTRO)::VARCHAR, DP.NO_PROFISSIONAL) AS NO_PROFISSIONAL,
    DP.NU_CNS,
    
    -- Cálculos
    1 AS QTD_DENOMINADOR,                             -- Denominador fixo para cálculos de cobertura
    COUNT(FAI.CO_SEQ_FAT_ATD_IND) AS QTD_ATENDIMENTOS -- Quantidade de atendimentos por grupo

FROM TB_FAT_ATENDIMENTO_INDIVIDUAL FAI

-- JOINs com dimensões
LEFT JOIN TB_DIM_TEMPO DT 
    ON DT.CO_SEQ_DIM_TEMPO = FAI.CO_DIM_TEMPO

LEFT JOIN TB_DIM_UNIDADE_SAUDE DUS 
    ON DUS.CO_SEQ_DIM_UNIDADE_SAUDE = FAI.CO_DIM_UNIDADE_SAUDE_1

LEFT JOIN TB_UNIDADE_SAUDE U 
    ON U.NU_CNES = DUS.NU_CNES

LEFT JOIN TB_DIM_PROFISSIONAL DP 
    ON DP.CO_SEQ_DIM_PROFISSIONAL = FAI.CO_DIM_PROFISSIONAL_1

LEFT JOIN TB_PROF_HISTORICO_CNS HC 
    ON HC.NU_CNS = DP.NU_CNS

LEFT JOIN TB_PROF TP 
    ON TP.CO_SEQ_PROF = HC.CO_PROF

LEFT JOIN TB_DIM_CBO DC 
    ON DC.CO_SEQ_DIM_CBO = FAI.CO_DIM_CBO_1

LEFT JOIN TB_DIM_EQUIPE DE 
    ON DE.CO_SEQ_DIM_EQUIPE = FAI.CO_DIM_EQUIPE_1

LEFT JOIN TB_EQUIPE E 
    ON E.NU_INE = DE.NU_INE 
   AND E.CO_UNIDADE_SAUDE = U.CO_SEQ_UNIDADE_SAUDE

LEFT JOIN TB_TIPO_EQUIPE TE 
    ON TE.CO_SEQ_TIPO_EQUIPE = E.TP_EQUIPE

-- Filtros
WHERE 
    FAI.CO_DIM_TEMPO BETWEEN '20260101' AND '20260430'
    AND TE.NU_MS IN ('70','76')
    AND DC.NU_CBO IN ('225142','225170','225130','223565','223505')
    AND FAI.CO_DIM_TIPO_ATENDIMENTO IN (2,3,4,5,6,7)

-- Agrupamento
GROUP BY 
    FAI.CO_DIM_UNIDADE_SAUDE_1,
    DUS.NU_CNES,
    DUS.NO_UNIDADE_SAUDE,
    FAI.CO_DIM_EQUIPE_1,
    DE.NU_INE,
    DE.NO_EQUIPE,
    FAI.CO_DIM_PROFISSIONAL_1,
    NU_PROFISSIONAL,
    NO_PROFISSIONAL,
    DP.NU_CNS,
    QTD_DENOMINADOR

ORDER BY 
    FAI.CO_DIM_UNIDADE_SAUDE_1,
    DUS.NU_CNES,
    FAI.CO_DIM_EQUIPE_1,
    NU_PROFISSIONAL;
