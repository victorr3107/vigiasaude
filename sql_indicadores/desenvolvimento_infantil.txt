SELECT 
    FCI.CO_FAT_CIDADAO_PEC AS CO_PRONTUARIO,
    CV.NO_CIDADAO,
    CV.NU_CNS_CIDADAO,
    CV.NU_CPF_CIDADAO,
    CV.NU_TELEFONE_CELULAR,
    CV.DT_NASCIMENTO_CIDADAO AS DT_NASCIMENTO,
    CV.NU_MICRO_AREA_TB_CIDADAO AS NU_MICRO_AREA_VINC,
    FCI.NU_MICRO_AREA AS NU_MICRO_AREA_FCI,
    CV.NU_CNES_VINC_EQUIPE AS NU_CNES_VINC,
    DUSI.NU_CNES AS NU_CNES_FCI,
    DUSV.NO_UNIDADE_SAUDE AS NO_UNIDADE_SAUDE_VINC,
    DUSI.NO_UNIDADE_SAUDE AS NO_UNIDADE_SAUDE_FCI,
    CV.NU_INE_VINC_EQUIPE AS NU_INE_VINC,
    DEI.NU_INE AS NU_INE_FCI,
    CAST(
        CASE 
            WHEN FCI.CO_DIM_SEXO = 1 THEN 'M'
            WHEN FCI.CO_DIM_SEXO = 2 THEN 'F'
            ELSE ''
        END AS CHAR(1)
    ) AS SEXO,
    COALESCE(TPI.CO_SEQ_PROF, DPI.CO_SEQ_DIM_PROFISSIONAL) AS NU_PROFISSIONAL_FCI,
    COALESCE(UPPER(TPI.NO_PROFISSIONAL_FILTRO)::VARCHAR, DPI.NO_PROFISSIONAL) AS NO_PROFISSIONAL_FCI,
    CAST(1 AS INTEGER) AS QTD_DENOMINADOR
FROM TB_ACOMP_CIDADAOS_VINCULADOS CV
LEFT JOIN TB_FAT_CAD_INDIVIDUAL FCI 
    ON FCI.NU_UUID_FICHA = CV.CO_UNICO_ULTIMA_FICHA
LEFT JOIN TB_DIM_TEMPO DT 
    ON DT.CO_SEQ_DIM_TEMPO = FCI.CO_DIM_TEMPO
LEFT JOIN TB_FAT_CIDADAO_PEC FCP 
    ON FCP.CO_SEQ_FAT_CIDADAO_PEC = FCI.CO_FAT_CIDADAO_PEC
LEFT JOIN TB_DIM_TEMPO DTN 
    ON DTN.CO_SEQ_DIM_TEMPO = FCP.CO_DIM_TEMPO_NASCIMENTO
LEFT JOIN TB_DIM_UNIDADE_SAUDE DUSV 
    ON DUSV.NU_CNES = CV.NU_CNES_VINC_EQUIPE
LEFT JOIN TB_DIM_UNIDADE_SAUDE DUSI 
    ON DUSI.CO_SEQ_DIM_UNIDADE_SAUDE = FCI.CO_DIM_UNIDADE_SAUDE
LEFT JOIN TB_DIM_EQUIPE DEI 
    ON DEI.CO_SEQ_DIM_EQUIPE = FCI.CO_DIM_EQUIPE
LEFT JOIN TB_DIM_PROFISSIONAL DPI 
    ON DPI.CO_SEQ_DIM_PROFISSIONAL = FCI.CO_DIM_PROFISSIONAL
LEFT JOIN TB_PROF_HISTORICO_CNS HC 
    ON HC.NU_CNS = DPI.NU_CNS
LEFT JOIN TB_PROF TPI 
    ON TPI.CO_SEQ_PROF = HC.CO_PROF
WHERE 
    CV.ST_POSSUI_FCI = 1
    AND COALESCE(FCI.CO_DIM_TIPO_SAIDA_CADASTRO, 3) = 3
    AND DTN.DT_REGISTRO::DATE >= '01/01/2024'
    AND DTN.DT_REGISTRO::DATE + INTERVAL '2 YEAR' <= '30/04/2028'
ORDER BY 
    CV.NO_CIDADAO,
    FCI.CO_FAT_CIDADAO_PEC,
    DT.DT_REGISTRO DESC;


SELECT COUNT(*) AS QTD_CONSULTAS
FROM TB_FAT_ATENDIMENTO_INDIVIDUAL AI
LEFT JOIN TB_DIM_TEMPO DT 
    ON DT.CO_SEQ_DIM_TEMPO = AI.CO_DIM_TEMPO
LEFT JOIN TB_DIM_UNIDADE_SAUDE DUS 
    ON DUS.CO_SEQ_DIM_UNIDADE_SAUDE = AI.CO_DIM_UNIDADE_SAUDE_1
LEFT JOIN TB_UNIDADE_SAUDE U 
    ON U.NU_CNES = DUS.NU_CNES
LEFT JOIN TB_DIM_CBO DC 
    ON DC.CO_SEQ_DIM_CBO = AI.CO_DIM_CBO_1
LEFT JOIN TB_DIM_EQUIPE DE 
    ON DE.CO_SEQ_DIM_EQUIPE = AI.CO_DIM_EQUIPE_1
LEFT JOIN TB_EQUIPE E 
    ON E.NU_INE = DE.NU_INE 
    AND E.CO_UNIDADE_SAUDE = U.CO_SEQ_UNIDADE_SAUDE
LEFT JOIN TB_TIPO_EQUIPE TE 
    ON TE.CO_SEQ_TIPO_EQUIPE = E.TP_EQUIPE
WHERE 
    AI.CO_FAT_CIDADAO_PEC = 31803
    AND (
        DC.NU_CBO LIKE '2235%' OR
        DC.NU_CBO LIKE '2231%' OR
        DC.NU_CBO LIKE '2251%' OR
        DC.NU_CBO LIKE '2252%' OR
        DC.NU_CBO LIKE '2253%'
    )
    AND TE.NU_MS IN ('70','76')
    AND AI.CO_DIM_TEMPO BETWEEN '20241111' AND '20241211';

SELECT DISTINCT 'AA' AS TIPO, FP.CO_SEQ_FAT_PROCED_ATEND_PROCED AS CO_FAT_PROCEDIMENTO
FROM TB_FAT_PROCED_ATEND_PROCED FP
LEFT JOIN TB_DIM_TEMPO DT ON DT.CO_SEQ_DIM_TEMPO = FP.CO_DIM_TEMPO
LEFT JOIN TB_DIM_UNIDADE_SAUDE DUS ON DUS.CO_SEQ_DIM_UNIDADE_SAUDE = FP.CO_DIM_UNIDADE_SAUDE
LEFT JOIN TB_UNIDADE_SAUDE U ON U.NU_CNES = DUS.NU_CNES
LEFT JOIN TB_DIM_PROCEDIMENTO DP ON DP.CO_SEQ_DIM_PROCEDIMENTO = FP.CO_DIM_PROCEDIMENTO
LEFT JOIN TB_DIM_CBO DC ON DC.CO_SEQ_DIM_CBO = FP.CO_DIM_CBO
LEFT JOIN TB_DIM_EQUIPE DE ON DE.CO_SEQ_DIM_EQUIPE = FP.CO_DIM_EQUIPE
LEFT JOIN TB_EQUIPE E ON E.NU_INE = DE.NU_INE AND E.CO_UNIDADE_SAUDE = U.CO_SEQ_UNIDADE_SAUDE
LEFT JOIN TB_TIPO_EQUIPE TE ON TE.CO_SEQ_TIPO_EQUIPE = E.TP_EQUIPE
WHERE 
    FP.CO_FAT_CIDADAO_PEC = 31803
    AND TE.NU_MS IN ('70','76')
    AND FP.CO_DIM_TEMPO BETWEEN '20241111' AND '20261111'
    AND DP.CO_PROCED IN ('0101040024','0301010269')

UNION ALL

SELECT DISTINCT 'PA', PA1.CO_FAT_PROCEDIMENTO
FROM TB_FAT_PROCED_ATEND PA1
LEFT JOIN TB_FAT_PROCED_ATEND_PROCED FP1 ON FP1.CO_FAT_PROCEDIMENTO = PA1.CO_FAT_PROCEDIMENTO
LEFT JOIN TB_DIM_PROCEDIMENTO DP1 ON DP1.CO_SEQ_DIM_PROCEDIMENTO = FP1.CO_DIM_PROCEDIMENTO
LEFT JOIN TB_DIM_TEMPO DT1 ON DT1.CO_SEQ_DIM_TEMPO = FP1.CO_DIM_TEMPO
LEFT JOIN TB_DIM_UNIDADE_SAUDE DUS ON DUS.CO_SEQ_DIM_UNIDADE_SAUDE = FP1.CO_DIM_UNIDADE_SAUDE
LEFT JOIN TB_UNIDADE_SAUDE U ON U.NU_CNES = DUS.NU_CNES
LEFT JOIN TB_DIM_CBO DC ON DC.CO_SEQ_DIM_CBO = FP1.CO_DIM_CBO
LEFT JOIN TB_DIM_EQUIPE DE ON DE.CO_SEQ_DIM_EQUIPE = FP1.CO_DIM_EQUIPE
LEFT JOIN TB_EQUIPE E ON E.NU_INE = DE.NU_INE AND E.CO_UNIDADE_SAUDE = U.CO_SEQ_UNIDADE_SAUDE
LEFT JOIN TB_TIPO_EQUIPE TE ON TE.CO_SEQ_TIPO_EQUIPE = E.TP_EQUIPE
WHERE 
    FP1.CO_FAT_CIDADAO_PEC = 31803
    AND TE.NU_MS IN ('70','76')
    AND FP1.CO_DIM_TEMPO BETWEEN '20241111' AND '20261111'
    AND DP1.CO_PROCED IN ('0101040083', 'ABPG039','0101040075', 'ABPG038');
